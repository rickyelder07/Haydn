2. Use trackUIStore for order:

```tsx
const { trackOrder, reorderTracks, resetAllStates } = useTrackUIStore();
```

3. Initialize track order when project loads (if not already done):

```tsx
useEffect(() => {
  if (project && trackOrder.length === 0) {
    useTrackUIStore.getState().initializeOrder(project.tracks.length);
  }
}, [project]);
```

4. Set up sensors for keyboard accessibility:
```tsx
const sensors = useSensors(
  useSensor(PointerSensor),
  useSensor(KeyboardSensor, { coordinateGetter: sortableKeyboardCoordinates })
);
```

5. Handle drag end:
```tsx
function handleDragEnd(event: DragEndEvent) {
  const { active, over } = event;
  if (over && active.id !== over.id) {
    const oldIndex = trackOrder.indexOf(Number(active.id));
    const newIndex = trackOrder.indexOf(Number(over.id));
    reorderTracks(oldIndex, newIndex);
  }
}
```

6. Render with DndContext and SortableContext:
```tsx
<DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
  <SortableContext items={trackOrder.map(String)} strategy={verticalListSortingStrategy}>
    {trackOrder.map((originalIndex) => {
      const track = trackDisplayInfo[originalIndex];
      if (!track) return null;
      return (
        <TrackItem
          key={originalIndex}
          id={String(originalIndex)}
          track={track}
          trackIndex={originalIndex}
          isSelected={selectedTrackIndex === originalIndex}
          onSelect={() => selectTrack(originalIndex)}
        />
      );
    })}
  </SortableContext>
</DndContext>
```

7. Add global reset button above track list:
```tsx
<button onClick={resetAllStates} className="text-sm text-gray-500 hover:text-gray-700">
  Reset Mute/Solo
</button>
```
