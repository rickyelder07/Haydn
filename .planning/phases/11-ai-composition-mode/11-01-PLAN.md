**compositionToHaydnProject** converter — maps 1:1 to HaydnProject format:
```typescript
export function compositionToHaydnProject(comp: MidiComposition): HaydnProject {
  const ppq = 480; // GENERATED_PPQ
  const maxTick = Math.max(0, ...comp.tracks.flatMap(t => t.notes.map(n => n.ticks + n.durationTicks)));
  return {
    originalFileName: null,
    sourceFormat: 'new',
    metadata: {
      name: `${comp.genre} - ${comp.key} ${comp.scale}`,
      ppq,
      tempos: [{ ticks: 0, bpm: comp.tempo }],
      timeSignatures: [{ ticks: 0, numerator: 4, denominator: 4 }],
      keySignatures: [{ ticks: 0, key: comp.key, scale: comp.scale === 'major' ? 'major' : 'minor' }],
    },
    tracks: comp.tracks.map(t => ({
      name: t.name,
      channel: t.channel,
      instrumentNumber: t.instrumentNumber,
      instrumentName: getInstrumentName(t.instrumentNumber) || 'Unknown',
      notes: t.notes,
      controlChanges: [],
    })),
    durationTicks: maxTick,
  };
}
```

**State type:**
```typescript
export type AiCompositionPhase =
  | 'idle'
  | 'questioning'    // Waiting for GPT-4o clarify response
  | 'answering'      // Questions displayed, waiting for user answers
  | 'generating'     // Calling GPT-4o for MIDI JSON (attempt 1-3)
  | 'retrying'       // Validation failed, retrying
  | 'success'
  | 'fallback'       // Used template after 3 failed AI attempts
  | 'error';

export interface ClarifyingTurn {
  questions: string[];  // Up to 3 questions from GPT-4o
  answers: string[];    // One answer per question from user
}

export interface CumulativeTokenUsage {
  clarifyTokens: number;
  generateTokens: number;
  totalTokens: number;
  totalCost: number;  // USD
}

interface AiCompositionState {
  phase: AiCompositionPhase;
  originalPrompt: string;
  clarifyingTurn: ClarifyingTurn | null;  // Single clarify turn (all questions at once)
  generateAttempt: number;               // 0-based, max 3
  tokenUsage: CumulativeTokenUsage | null;
  costEstimate: TokenEstimate | null;    // Pre-generation estimate
  lastError: string | null;
  // Actions
  startClarification: (prompt: string) => Promise<void>;
  submitAnswers: (answers: string[]) => Promise<void>;
  skipQuestions: () => Promise<void>;
  reset: () => void;
}
```

**startClarification(prompt):**
1. Set phase='questioning', originalPrompt=prompt, clarifyingTurn=null, generateAttempt=0, tokenUsage=null, lastError=null
2. POST to `/api/ai-compose` with `{ prompt, clarifyingHistory: [] }` (no generateAttempt field = clarify mode)
3. If response has `questions.length > 0`: set phase='answering', clarifyingTurn={ questions, answers: [] }
4. If response has `questions.length === 0` (prompt was detailed enough): auto-advance by calling `_runGenerate(prompt, null)`

**submitAnswers(answers):**
1. Update clarifyingTurn.answers=answers
2. Re-estimate cost using full conversation text (system prompt approximation + prompt + Q&A), pass estimatedOutputSize=3000 to estimateTokensAndCost
3. Set costEstimate=result
4. Call `_runGenerate(originalPrompt, clarifyingTurn)`

**skipQuestions():**
1. Set clarifyingTurn=null
2. Estimate cost with just the prompt (estimatedOutputSize=3000)
3. Set costEstimate=result
4. Call `_runGenerate(originalPrompt, null)`

**_runGenerate(prompt, turn) — internal async helper (not in state, just a local function):**
```
for attempt 0..2:
  set phase='generating' (attempt 0) or 'retrying' (attempt > 0), generateAttempt=attempt
  POST /api/ai-compose with { prompt, clarifyingHistory: buildHistory(turn), generateAttempt: attempt, validationErrors: previousErrors }
  on success:
    convert response.data to HaydnProject via compositionToHaydnProject()
    validate: for each non-drum track, for each note, run pipeline.validate()
    collect unique error messages (errors only, not warnings — follow nlGenerationStore pattern)
    if errors.length === 0:
      accumulate tokenUsage from response.usage (add to clarify tokens)
      call useProjectStore.getState().loadNewProject(project)
      set phase='success', tokenUsage=cumulative
      return
    else:
      previousErrors = unique error messages
      continue loop
// After 3 failed attempts:
set phase='fallback'
const genre = extractGenreFromPrompt(prompt) — detect 'lofi'|'trap'|'boom-bap'|'jazz'|'classical'|'pop' from prompt text, default 'pop'
const fallbackParams = { genre, tempo: 90, key: 'C', scale: 'minor', ... sensible defaults }
const project = assembleProject(fallbackParams as GenerationParams)
useProjectStore.getState().loadNewProject(project)
```
